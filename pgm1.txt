char* strrev(char* name){

int i1=0,l;
char temp;
					l=strlen(name);
					while(i1<l)
						{
							temp=name[i1];
							name[i1]=name[l-1];
							name[l-1]=temp;
							i1++;
							l--;
						}
					return name;
}
